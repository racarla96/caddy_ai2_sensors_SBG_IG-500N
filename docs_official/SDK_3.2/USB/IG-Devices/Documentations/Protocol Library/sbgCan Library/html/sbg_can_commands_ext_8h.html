#define SBG_CAN_EXT_PORT_RS232   (0x0000)

Set the external port in RS-232 mode (default)

#define SBG_CAN_EXT_PORT_RS422   (0x0001)

Set the external port in RS-422 mode

#define SBG_CAN_EXT_PORT_FAST_SLEW   (0x0000)

Fast slew rate, baudrate is not limited. (default)

#define SBG_CAN_EXT_PORT_SLOW_SLEW   (0x0002)

Slow slew rate for EMI reduction. Baudrate is limited to 230 400 bps.
