#define SBG_CAN_FILTER_OPTION_ENABLE_ATTITUDE   (0x10)

Enable the attitude/navigation computation.
If disabled, the device can outputs calibrated data at 512 Hz.

#define SBG_CAN_GPS_DISABLE_SBAS   (0x00)

GPS option, disable SBAS and ranging

#define SBG_CAN_GPS_ENABLE_SBAS_DIFF_CORRECTIONS   (0x01)

GPS option, enable SBAS corrections

#define SBG_CAN_GPS_ENABLE_SBAS_RANGING   (0x02)

GPS option, enable ranging for SBAS
