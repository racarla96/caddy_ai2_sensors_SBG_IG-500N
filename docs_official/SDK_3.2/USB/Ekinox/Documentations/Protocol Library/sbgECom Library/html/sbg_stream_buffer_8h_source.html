175 SBG_INLINE SbgErrorCode sbgStreamBufferSeek(SbgStreamBuffer *pHandle, uint32 offset, SbgSBSeekOrigin origin)
176 {
177  SbgErrorCode errorCode = SBG_NO_ERROR;
178 
179  //
180  // TODO: test for buffer overflow
181  //
182 
183  //
184  // According to the origin reference point
185  //
186  switch (origin)
187  {
188  case SB_SEEK_SET:
189  pHandle->pCurrentPtr = pHandle->pBufferPtr + offset;
190  break;
191  case SB_SEEK_CUR_INC:
192  pHandle->pCurrentPtr += offset;
193  break;
194  case SB_SEEK_CUR_DEC:
195  pHandle->pCurrentPtr -= offset;
196  break;
197  case SB_SEEK_END:
198  pHandle->pCurrentPtr = pHandle->pBufferPtr + (pHandle->bufferSize - offset);
199  break;
200  default:
201  errorCode = SBG_INVALID_PARAMETER;
202  }
203 
204  return errorCode;
205 }
